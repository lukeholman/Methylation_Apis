Genomic context of sites with caste-specific methylation

The following uses the results of the GLMs generated below, which estimates the difference in mean % methylation for QDL and WDL at each time point, for each site (as well as the standard error associated with this difference in means). These can be thought of as effect sizes, which describe the effect of caste on methylation at each site.

In this section, we analyse those effect sizes using a linear model, in order to estimate the average effect size for sites of each type. This was accomplished by running a linear model with the following formula:

\[y_i = \alpha + \beta_1x_{1i} + \beta_1x_{2i} + \beta_1x_{3i} + \beta_1x_{4i} + \beta_1x_{5i} + \epsilon_i\] Here, \(y_i\) is the effect size for site \(i\), \(\alpha\) is the global intercept, the five \(\beta\) terms are regression coefficients to be estimated, the five \(x_i\) terms are binary predictor variables for site \(i\), and \(\epsilon_i\) is the residual (assumed to follow a Normal distribution). The five binary predictor variables describe whether the focal site is (or is not) in a coding region, a promoter, an intron, a 5’-UTR, or a 3’-UTR respectively; note that any one site can overlap more than one of these genomic features. Promoter regions were defined as regions that are up to 2000bp upstream or 200bp downstream of a gene. After fitting the model, we then used the estimated regression coefficients to find the estimated average effect size for a site that maps uniquely to each one of the five genomic features. One model was run for each time point (i.e. one model for each of the figure panels below).

Run one GLM per site using base::glm()

Here, we analyse the 112001 methylated sites for which we have sufficient data using one generalised linear model per site, via the function base::glm().

To summarise the following code:

  • These GLMs include all 36 samples where possible. Samples were omitted when the focal site was covered by <4 reads in the focal sample.
  • Non-linear effects of Time were modeled using a quadratic term, allowing accelerating or deccelerating changes in % methylation with time
  • For each site, we fit 6 different GLMs that we considered to be biologically plausible, and then performed model averaging based on their Akaike weights using MuMIn::model.avg(). All 6 models contained experimental block (rep in the models below) as a fixed effect, and optionally also contained the fixed effects time, time^2, the caste:time interaction, and the I(time^2):caste interaction. Note that we did not fit caste as predictor except in interactions with time, which forces queens and workers to have the same estimated % methylation at time zero (which we know must be the case due to our experimental design)
  • The response variable was the number of cytosines observed at the site, given the total coverage at the sites (i.e. number of reads containing Cs plus Ts). We used Binomial errors and a logit link.
  • After fitting the models and averaging them, we predicted the % methylation of the site in queens and workers at each time point, and calculated the caste difference, its 95% CIs, and the p-value for the null hypothesis that the caste difference was zero using bootstrapping.
  • We also saved the p-value for caste:time or I(time^2):caste (whichever was greater) from the top-ranked model, using the car::Anova() function to run a likelihood ratio test. We use this to score the ‘significance’ of caste for each site, and this value is presented in the table of results below (-log10 transformed)

# Calculate AIC weight from delta-AIC values
# The sum of the weights for all models containing a given predictor gives its "variable importance" 
get_aic_weight <- function(aic_deltas){
  exp(-0.5 * aic_deltas) / sum(exp(-0.5 * aic_deltas))
}

# inverse logit to convert from the logit link scale to a proportion
inv_logit <- function(x) 1 / (1 + exp(-x))

# run a binomial GLM given a formula and some data
run_glm <- function(my_formula, dat) {
  glm(as.formula(my_formula), data = dat, family = "binomial")
}

# Get random normals for nSamples, nBoots times, for each value in a vector of means and SDs
# means and SDs should be the same length. Returns an array
# Used below for bootstrapping the caste difference in methylation rate at each time point
vectorised_rnorm <- function(nSamples, nBoots, means, SDs) {
  m <- length(means)
  aperm(array(rnorm(nSamples * m * nBoots, means, SDs), 
              c(m, nSamples, nBoots)), c(2,1,3))
}

# average the models, and save key results. The function either uses the real data, or shuffles them
model_multiple_sites <- function(chunk_number, 
                                 vector_of_chunks, 
                                 make_plot_data = FALSE){
  
  print("Starting new chunk")
  focal_sites <- vector_of_chunks[[chunk_number]]
  
  # Get data from the database for the focal chunk of sites
  focal_chunk <- by_site %>%  
    filter(site %in% focal_sites) %>% 
    dplyr::select(sample, site, numC, numT, caste, time, rep) %>%
    distinct() %>%
    collect() %>%
    mutate(caste = replace(caste, caste == "t", "Worker")) # t0 caste labelled as worker
  
  # Internal function to model one site using GLM, and compare models by AICc
  do_one_site <- function(focal_site, data_chunk, nboots = 10000){
    
    focal_site_data <- data_chunk %>%
      filter(site == focal_site)
    
    models <- lapply(formulae, run_glm, dat = focal_site_data)
    
    AICc_table <- data.frame(
      Site = focal_site,
      formula = str_remove_all(formulae, "cbind[(]numC, numT[)] ~ "),
      model = 1:length(models),
      AICc(models[[1]], models[[2]], models[[3]], models[[4]], models[[5]], models[[6]])
    ) %>% as_tibble() %>% 
      arrange(AICc) %>%
      mutate(delta = AICc - AICc[1],
             weight = get_aic_weight(delta))
    
    # Get the anova table for top-AICc model
    top_model_anova <- as.data.frame(car::Anova(models[[AICc_table$model[1]]], type = "III")) %>%
      rownames_to_column("Parameter") %>%
      mutate(Site = focal_site)
    AICc_table <- AICc_table %>% dplyr::select(-model)
    
    # Predict new data and calculate caste difference, using model average parameter estimates
    new <- expand_grid(caste = c("Queen", "Worker"), 
                       time = c(0, 2, 4, 6, 8), 
                       rep = c("A", "B", "C", "D"))
    
    
    # Note: predict on the link scale, not data scale
    predictions_with_se <-
      data.frame(new,
                 predicted_prop_meth = predict(
                   model.avg(models), backtransform = FALSE,
                   newdata = new, se.fit = TRUE)) %>%
      group_by(caste, time) %>%
      summarise(predicted_prop_meth.fit = mean(predicted_prop_meth.fit),
                predicted_prop_meth.se.fit = mean(predicted_prop_meth.se.fit), .groups = "drop")
    
    # Get the caste mean prop. methylation (note back-transform from the link scale)
    caste_means <- predictions_with_se %>% 
      dplyr::select(-predicted_prop_meth.se.fit) %>% 
      spread(caste, predicted_prop_meth.fit) %>%
      mutate(Site = focal_site,
             Queen = inv_logit(Queen),
             Worker = inv_logit(Worker),

process_GLM_output <- function(file_path){
  all_chunks <- 
    list.files(file_path, full.names = TRUE) %>%
    map(~ readRDS(.x))
  
  caste_importance <- all_chunks %>% 
    map_df(~ .x[[1]]) %>%
    filter(str_detect(formula, "caste")) %>%
    group_by(Site) %>%  # sum the weights over the models containing CASTE, i.e. "variable importance"
    summarise(caste_importance = sum(weight), .groups = "drop")
  
  time_importance <- all_chunks %>% 
    map_df(~ .x[[1]]) %>%
    filter(str_detect(formula, "time")) %>%
    group_by(Site) %>%  # sum the weights over the models containing TIME, i.e. "variable importance"
    summarise(time_importance = sum(weight), .groups = "drop")
  
  interaction_importance <- all_chunks %>% 
    map_df(~ .x[[1]]) %>%
    filter(str_detect(formula, "caste[:]time")) %>%
    group_by(Site) %>%  # sum the weights over the models containing CASTE, i.e. "variable importance"
    summarise(interaction_importance = sum(weight), .groups = "drop")
  
  importance <- caste_importance %>%
    left_join(time_importance, by = "Site") %>%
    left_join(interaction_importance, by = "Site") %>%
    arrange(-interaction_importance)
  
  # P value for the most signficant parameter involving caste, from the top-AICc model
  # Only sites where the top model contained caste appear here (others assumed < 0.05)
  # The p value is either for the caste:time or caste:time^2 effect, whichever is lesser
  p_values <- all_chunks %>% 
    map_df(~ .x[[3]]) %>%
    filter(str_detect(Parameter, "caste")) %>%
    group_by(Site) %>%
    summarise(p = min(`Pr(>Chisq)`), .groups = "drop") 
  
  map_df(all_chunks, ~ .x[[2]]) %>%
    write_tsv(path = "output/frequentist_GLM_meth_diff.txt")

GO and KEGG enrichment of genes showing variable methylation

The results in the following plots were calculated using GSEA tests for enrichment, implemented in the fgsea package. This test involves ranking genes by some statistic and then searching for GO and KEGG terms that are over-represented among genes at the top or bottom of the list. Here, we ranked the genes by the average of the difference in % methylation between ‘classes’ that was estimated for all the sites in the gene (defined as sites inside the exons, introns, or promoter regions of the gene). ‘Classes’ either means queens and workers (measured at the 8h time point), or the 0h and 8h time points (measured in either queens or workers). We derive these estimates by model averaging GLMs that were run on each of the 112027 sites for which sufficient data were available.

The output of fgsea includes the “normalised enrichment score” (NES) which describes how much the pathway is over-represented at the top of this list (for positive NES), or over-represented at the bottom (negative NES), and there is an associated \(p\)-value for each NES (calculated by resampling).

To create a co-methylation network, we used the % methylation values per gene (i.e. pooling across all the CpG sites in each gene) that were inferred by the BWASP pipeline. We analysed all genes that showed non-zero variability in % methylation across samples, which had >=2.9% mean methylation frequency, and which were methylated in >=30 samples. The dataset used to build the co-methylation network thus contained 36 samples and 5910 genes.

We then used WGCNA::blockwiseModules() to create a signed gene co-methylation network, largely using the default settings.

Fit Bayesian model to the eigengene estimates

The following code chunk fits a Bayesian multivariate model to the eigengene estimates from the co-methylation network produced by WGCNA. The ‘eigengene’ for sample \(i\), module \(j\), gives a measure of the methylation level of all the co-methylated genes that were assigned to module \(j\), for sample \(i\) relative to other samples. The eigengenes were transformed to have a mean of zero and standard deviation of 1 prior to analysis.

Because the module eigengenes are somewhat correlated with each other, it is valuable to run a single model of all \(j\) module eigengenes (thereby using the information in this correlation), rather than running \(j\) univariate models. We therefore ran a single multivariate model encompassing all 9 modules (including module zero, the unassigned genes).

Furthermore, we expected a priori that some eigengenes will change non-linearly, for example by increasing to a plateau. We therefore used the same quadratic formula (implemented with `brms’) to model the eigengenes, to allow for a non-linear (and potentially caste-specific) time effect. As before, we ran the model on different versions of the data where the \(t_0\) samples were labeled as either queens or workers in each possible balanced combination then combined the posterior parameter estimates; this procedure ensures that the \(t_0\) samples have an equal effect on the predicted values for queens and workers. Note that the main effect of caste is again missing from the model, which forces the queen and worker samples to have the same value at \(t_0\)

In brms notations (which is similar to e.g. the lme4 and mgcv R packages), the model formula for each module is written as

\[ y \sim Caste:Time + I(Time^2) + I(Time^2):Caste + Replicate\]

where \(y\) is the eigengene for module \(j\).

    p <- 1 - (as.numeric(bayestestR::p_direction(x)))

      mutate(PP = p) %>%
      mutate(` ` = ifelse(PP < 0.1, "~", ""),

             ` ` = replace(` `, PP < 0.01, "**"),
             ` ` = replace(` `, PP < 0.001, "***"),

}

# Code up a hypergeometric test
# Inspiration for this code: https://stats.stackexchange.com/questions/10328/using-rs-phyper-to-get-the-probability-of-list-overlap
manual_GO_enrich <- function(
  GO_ID,
  number_genes_in_set_with_focal_GO,
  number_genes_in_universe_with_focal_GO,
  number_genes_in_set,
  number_genes_in_universe){
  
  p <- phyper(
    q = number_genes_in_set_with_focal_GO - 1, 
    m = number_genes_in_universe_with_focal_GO, 
    n = number_genes_in_universe - number_genes_in_universe_with_focal_GO, 
    k = number_genes_in_set, 
    
    lower.tail = FALSE, log.p = FALSE)
  
  prop_in_set  <- number_genes_in_set_with_focal_GO / number_genes_in_set
  prop_in_univ <- number_genes_in_universe_with_focal_GO / number_genes_in_universe
  
  # Log odds ratio: odds that the focal GO is in the set / odds it's in the universe
  enrichment_LOR <- (prop_in_set / (1 - prop_in_set)) /
    (prop_in_univ / (1 - prop_in_univ))
  
  tibble(
    ID = GO_ID,
    GeneRatio = paste(number_genes_in_set_with_focal_GO, number_genes_in_set, sep="/"), 
    BgRatio = paste(number_genes_in_universe_with_focal_GO, number_genes_in_universe, sep="/"),
    enrichment_LOR,
    pvalue = p
  )
}

  apis.db, GO_list, 
  min_size, keep_all){

  # Define internal GO enrichment function (custom; gives same results as clusterProfiler)
  # However, it works without an Org.db, which we needed in order to use our custom GO annotations
  GO_enrichment <- function(gene_set,
                            gene_universe,
                            min_size,
                            keep_all){

    # get GO annotations for genes in the focal set/module
    gene_set <- tbl(db, GO_list) %>%
      left_join(tbl(db, "gene_names"), 
                by = c("SYMBOL" = "gene_symbol")) %>%
      dplyr::rename(gene = SYMBOL) %>%
      dplyr::select(gene, GO, gene_name) %>%
      filter(gene %in% gene_set) %>%
      collect(n=Inf) %>%
      distinct() # should already be, add this to double check
    
    #  do the same for the gene universe
    gene_universe <- tbl(db, GO_list) %>%
      left_join(tbl(db, "gene_names"), 
                by = c("SYMBOL" = "gene_symbol")) %>%
      dplyr::rename(gene = SYMBOL) %>%
      dplyr::select(gene, GO, gene_name) %>%
      filter(gene %in% gene_universe) %>%
      collect(n=Inf) %>%
      distinct() # should already be, add this to double check
    
    unique_GOs <- unique(gene_set$GO)

    universe_size <- nrow(gene_universe)

      if(x1 < min_size) return(NULL)

        GO_ID = focal_GO,
        number_genes_in_set_with_focal_GO = x1,
        number_genes_in_universe_with_focal_GO = x2,
        number_genes_in_set = module_size,
        number_genes_in_universe = universe_size
      )
    }) 
    
    if(!keep_all & nrow(output) > 0) output <- output %>% filter(pvalue < 0.05)
    if(nrow(output)==0) return(NULL)
    
    output %>%
      left_join(tbl(db, "go_meanings") %>% collect(),
                by = c("ID" = "GO")) %>%
      dplyr::rename(Ontology = ontology,
                    Meaning = term) %>%
      dplyr::select(Ontology, ID, Meaning, everything()) %>%
      distinct()
  }

  kegg_enrichment <- function(gene_set, gene_universe, 
                              GO_list, min_size, keep_all){
    
    # if using D.mel annotations, first convert among orthologuous gene names
    if(GO_list == "dros_ortho_GO"){

        pull(FLYBASECG) %>% unique()

        left_join(tbl(db, "dros_ortho_GO") %>% 
                    dplyr::select(FLYBASECG, SYMBOL) %>% collect(n=Inf),
                  by = "SYMBOL") %>%
        filter(!is.na(FLYBASECG)) %>%
        pull(FLYBASECG) %>% unique()

      gene_universe_CG_drosophila <- paste("Dmel_", gene_universe_CG_drosophila, sep = "")

    }
    
    if(is.null(result)) return(NULL) # if no enriched/annotated terms...
    
    result <- result@result
    is_enriched <- sapply(result$GeneRatio, 
                          function(x) eval(parse(text = x))) > 
      sapply(result$BgRatio, 
             function(x) eval(parse(text = x)))
    result <- result[is_enriched, ] %>% 
      mutate(p.adjust = p.adjust(pvalue, method = "BH"))
    
    if(nrow(result %>% filter(pvalue < p)) == 0) return(NULL)
    if(!keep_all) result <- result %>% filter(pvalue < 0.05)

  }

                              function(x) x[1] / x[2]),

      dplyr::select(Ontology, ID, Meaning, GeneRatio, BgRatio, enrichment_LOR, pvalue) 
  }
  
  kegg_result %>%
    bind_rows(GO_enrichment(gene_set, gene_universe, min_size, keep_all))
}
